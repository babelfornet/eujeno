# CLI `eujeno mcp` + `infer --mcp` — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development. Steps use checkbox (`- [ ]`).

**Goal:** Configure and use **MCP tools from the command line**: `eujeno mcp` (add/list/remove of MCP servers, persisted) and `eujeno infer --mcp` (queries the model, running the tool-calling loop with the configured MCP tools). This way an AI with shell access (e.g. Claude Code) can also do #4 via the CLI.

**Architecture:** MCP config persisted in `~/.eujeno/mcp.json` (override with `EUJENO_HOME`). The `mcp` command (single word + switches, consistent with `model --info`) manages the config and, for `--list`, discovers the tools via `McpRegistry`. `infer --mcp` builds an `McpRegistry` from the config, fetches the tools, and runs `run_tool_loop` against the `/v1/chat/completions` endpoint of the coordinator or a peer (both expose it), executing the tools via MCP. Reuses everything that already exists (`McpRegistry`, `run_tool_loop`).

**Tech Stack:** Python · Typer · the existing `eujeno/ui/{mcp,agent}.py` · httpx.

**Out of scope:** MCP via SSE/HTTP (stdio only); streaming.

```python
import json
import os


def _home() -> str:
    return os.environ.get("EUJENO_HOME") or os.path.join(os.path.expanduser("~"), ".eujeno")


def _path() -> str:
    return os.path.join(_home(), "mcp.json")


def load_servers() -> dict:
    p = _path()
    if not os.path.exists(p):
        return {}
    try:
        with open(p, encoding="utf-8") as f:
            return json.load(f).get("servers", {})
    except Exception:
        return {}


def save_servers(servers: dict) -> None:
    os.makedirs(_home(), exist_ok=True)
    with open(_path(), "w", encoding="utf-8") as f:
        json.dump({"servers": servers}, f, indent=2)


def add_server(name: str, command: str, args=None) -> dict:
    servers = load_servers()
    servers[name] = {"command": command, "args": list(args or [])}
    save_servers(servers)
    return servers


def remove_server(name: str) -> dict:
    servers = load_servers()
    servers.pop(name, None)
    save_servers(servers)
    return servers
```

@app.command()
def mcp(
    add: str = typer.Option(None, "--add", help="Name of an MCP server to add"),
    command: str = typer.Option(None, "--command", help="MCP server command (with --add)"),
    args: str = typer.Option(None, "--args", help="Command arguments, space-separated"),
    remove: str = typer.Option(None, "--remove", help="Name of an MCP server to remove"),
):
    """Configure the MCP servers (tools) usable by 'eujeno infer --mcp'. Without switches: list them."""
    from eujeno.mcp_config import load_servers, add_server, remove_server
    if add:
        if not command:
            _fail("mcp", "USAGE_ERROR", "--command is required with --add", exit_code=2)
        servers = add_server(add, command, (args or "").split())
        _emit_ok("mcp", {"servers": list(servers.keys())}, human=f"added MCP server: {add}")
        return
    if remove:
        servers = remove_server(remove)
        _emit_ok("mcp", {"servers": list(servers.keys())}, human=f"removed: {remove}")
        return
    servers = load_servers()
    tools = []
    if servers:
        from eujeno.ui.mcp import McpRegistry
        reg = McpRegistry()
        for name, cfg in servers.items():
            reg.add(name, cfg["command"], cfg.get("args", []))
        try:
            tools = [{"name": t["function"]["name"], "description": t["function"]["description"]}
                     for t in reg.list_tools()]
        except Exception as e:
            _emit_ok("mcp", {"servers": list(servers.keys()), "tools": [], "error": str(e)},
                     human=f"servers: {list(servers.keys())}  (tools not listable: {e})")
            return
    human = "\n".join([f"MCP servers: {list(servers.keys())}"] + [f"  🔧 {t['name']} — {t['description']}" for t in tools])
    _emit_ok("mcp", {"servers": list(servers.keys()), "tools": tools}, human=human or "no MCP server configured")
```

@pytest.mark.slow
def test_infer_mcp_runs_tool_loop(monkeypatch, tmp_path):
    import socket, threading, time
    import uvicorn
    from fastapi import FastAPI, Request

    monkeypatch.setenv("EUJENO_HOME", str(tmp_path))
    runner.invoke(app, ["mcp", "--add", "echo", "--command", sys.executable, "--args", _ECHO])

    # fake /v1: first round requests the echo__echo tool, second round replies
    state = {"n": 0}
    fake = FastAPI()

    @fake.post("/v1/chat/completions")
    async def v1(request: Request):
        state["n"] += 1
        if state["n"] == 1:
            return {"choices": [{"message": {"role": "assistant", "content": None,
                    "tool_calls": [{"id": "c0", "type": "function",
                                    "function": {"name": "echo__echo", "arguments": json.dumps({"text": "hi"})}}]},
                    "finish_reason": "tool_calls"}]}
        return {"choices": [{"message": {"role": "assistant", "content": "The tool replied."},
                "finish_reason": "stop"}]}

    s = socket.socket(); s.bind(("127.0.0.1", 0)); port = s.getsockname()[1]; s.close()
    srv = uvicorn.Server(uvicorn.Config(fake, host="127.0.0.1", port=port, log_level="error"))
    threading.Thread(target=srv.run, daemon=True).start()
    for _ in range(200):
        if srv.started: break
        time.sleep(0.05)
    try:
        r = runner.invoke(app, ["--json", "infer", "--coordinator", f"http://127.0.0.1:{port}",
                                "--mcp", "--prompt", "use echo"])
        assert r.exit_code == 0, r.stdout
        data = json.loads(r.stdout)["data"]
        assert data["text"] == "The tool replied."
        assert any(run["name"] == "echo__echo" for run in data.get("tool_runs", []))
    finally:
        srv.should_exit = True
```
